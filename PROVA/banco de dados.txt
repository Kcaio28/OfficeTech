-- ENTREGA 3: SCRIPT DE CRIAÇÃO E POPULAÇÃO DO BANCO DE DADOS

DROP DATABASE IF EXISTS officetech_db;
CREATE DATABASE officetech_db;
USE officetech_db;

-- 2. CRIAÇÃO DA TABELA USUARIO
CREATE TABLE clientes (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(50) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL
);

-- 3. CRIAÇÃO DA TABELA PRODUTOs
CREATE TABLE equipamento (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT,
	estado ENUM('Disponivel', 'Alugado')
);

-- 4. CRIAÇÃO DA TABELA MOVIMENTACAO (RF14 - Histórico/Rastreabilidade)
CREATE TABLE movimentacao (
    id INT PRIMARY KEY AUTO_INCREMENT,
    tipo_movimentacao ENUM('Devolução', 'Aluguel') NOT NULL, 
    data_saida DATE NOT NULL,
	data_devolucao_prevista DATE NOT NULL,
	data_devolucao_real DATE NOT NULL,
    
    -- Chaves Estrangeiras para Rastreabilidade
    id_equipamento INT,
    id_cliente INT, -- Responsável pela movimentação
    
    FOREIGN KEY (id_equipamento) REFERENCES equipamento(id),
    FOREIGN KEY (id_cliente) REFERENCES clientes(id)
);
